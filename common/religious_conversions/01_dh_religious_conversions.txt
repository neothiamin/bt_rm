cathar_center_of_reformation = {
	
	religion = cathar

	map_icon = GFX_religious_conversion_center_protestant

	max_distance = 100.0 #Distance from CoR province. Vanilla CoRs are all 150, but Cathar CoRs are pretty much all in/near France, so the lower distance seems ok.

	base_conversion_speed = 1.5 #3/4 of vanilla. I don't want Cathar spreading like wildfire, especially because with the centrality of its CoRs it can dominate the France region quickly. Plus Cathar gets events to spread it, like the Mass Conversions in Toulouse event and the Revivalists in Lombardy event. Note: in-game, this value rounds down to 1 visually.

	target_province_weights = {
		factor = 2
		
		modifier = {
			factor = 0
			is_empty = yes
		}
		modifier = {
			factor = 0
			has_missionary = yes
		}
		modifier = {
			factor = 0
			province_id = 118 # Rome, seat of Papacy
		}
		modifier = {
			factor = 0
			has_province_modifier = religious_center
		}
		modifier = {
			factor = 0
			has_province_modifier = religious_zeal_at_conv
		}
		modifier = {
			factor = 0
			religion = reformed
		}
		modifier = {
			factor = 0
			religion = protestant
		}
		modifier = {
			factor = 0
			NOT = { religion_group = christian }
		}
		modifier = {
			factor = 0
			NOT = { continent = FROM }
		}
		modifier = {
			factor = 0.6
			owner = {
				num_of_cardinals = 1
			}
		}
		modifier = {
			factor = 0.6
			owner = {
				num_of_cardinals = 2
			}
		}
		modifier = {
			factor = 0.6
			owner = {
				num_of_cardinals = 3
			}
		}
		modifier = {
			factor = 0.6
			owner = {
				num_of_cardinals = 4
			}
		}
		modifier = {
			factor = 0.6
			owner = {
				num_of_cardinals = 5
			}
		}
		#Whyyy!? Converted nations get bonuses anyways, and this effectively discourages spreading outside of Cathar nations, which is, you know, the opposite of what should happen.
		#modifier = {
		#	factor = 1.25
		#	owner = {
		#		religion = cathar
		#	}
		#}
		#Giving a 0.5x factor to different culture group provinces means that this is unnecessary.
		#modifier = {
		#	factor = 2
		#	culture_group = FROM
		#}
		modifier = {
			factor = 0.5
			NOT = {
				culture_group = FROM
			}
		}
		modifier = {
			factor = 2
			culture = FROM
		}
		modifier = {
			factor = 0.01
			religion = coptic
		}		
		modifier = {
			factor = 0.01
			religion = orthodox
		}
		modifier = {
			factor = 0.01
			religion = fraticelli
		}
		#This makes sense, I think. Of course France, especially Toulouse, was famous for its Cathars, but Cathar theology (according to some sources) may have started in the Rhineland.
		modifier = {
			factor = 1.5
			OR = {	
				area = north_rhine_area
				area = lower_rhineland_area
				area = westphalia_area			
				area = north_westphalia_area #Added in BT 13
				region = france_region
				#region = low_countries_region
			}
		}
		modifier = {
			factor = 1.1
			region = low_countries_region
		}
		modifier = {
			factor = 1.1
			has_port = yes
		}
		modifier = {
			factor = 0
			NOT = {
				calc_true_if = {
					amount = 50
					all_province = {
						religion = cathar
					}
				}
			}
			religion_years = {
				protestant = 20
			}
		}
	}
}

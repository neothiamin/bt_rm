dh_crazy_mode_cor = {	
	if = {
		limit = {
			has_global_flag = dh_crazy_mode
		}
		NOT = {
			num_of_centers_of_reformation = {
				religion = $RELIGION$
				count = $COUNT1$
			}
		}
	}
	else = {
		NOT = {
			num_of_centers_of_reformation = {
				religion = $RELIGION$
				count = $COUNT2$
			}
		}
	}
}

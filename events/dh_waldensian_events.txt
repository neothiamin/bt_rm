#Waldensian Mission Discovered in [Root.GetName]
province_event = {
	id = dh_waldo.1
	title = dh_waldo.1.t
	desc = dh_waldo.1.desc
	picture = PRAYING_eventPicture
	goto = root

	trigger = {
		dh_waldo_spawn_area = yes
		religion = catholic
		had_global_flag = {
			flag = dh_waldo_religion_enabled
			days = 1095
		}
		#development = 8
		NOT = {
			development = 20
		}
		is_reformation_center = no
		dh_can_have_heresy_cor_trigger = {
			RELIGION = waldensian
		}		
		NOT = {
			religion_years = {
				waldensian = 75
			}
		}
	}

	fire_only_once = yes

	mean_time_to_happen = {
		months = 60
		modifier = {
			factor = 0.5
			OR = {
				has_terrain = hills
				has_terrain = mountain
			}
		}
		modifier = {
			factor = 0.5
			NOT = {
				development = 12
			}
		}
	}

	major = yes

	#Surely they will keep to themselves, yes?
	option = {
		name = dh_waldo.1.a
		enable_religion = waldensian
		change_religion = waldensian
		add_reform_center = waldensian
		save_global_event_target_as = dh_waldo_spawn_prov
		dh_add_cor_resistance_to_theocracies = yes
		dh_add_neighbor_conv_weakness = yes
	}
}

#Waldensians from the Woodwork
province_event = {
	id = dh_waldo.2
	title = dh_waldo.2.t
	desc = dh_waldo.2.desc
	picture = PRAYING_eventPicture
	goto = root

	trigger = {
		dh_waldo_spread_area = yes
		religion = catholic
		#is_religion_enabled = waldensian
		religion_years = {
			waldensian = 3
		}
		dh_can_have_heresy_cor_trigger = {
			RELIGION = waldensian
		}
		if = {
			limit = {
				region = france_region
			}			
			development = 14
		}
		else_if = {
			limit = {
				region = italy_region
			}
			development = 18
		}
		else_if = {
			limit = {
				region = south_german_region
			}
			development = 30
		}
		OR = {
			NOT = {
				has_global_flag = dh_waldo_timer
			}
			had_global_flag = {
				flag = dh_waldo_timer
				days = 365
			}
		}
		NOT = {
			owner = {
				calc_true_if = {
					all_province = {
						religion = waldensian
						is_reformation_center = yes
					}
					amount = 3
				}
			}
			religion_years = {
				waldensian = 50
			}
			any_province_in_state = {
				AND = {
					religion = waldensian
					is_reformation_center = yes
				}
			}
			is_reformation_center = yes
		}
	}

	mean_time_to_happen = {
		years = 6
	}

	immediate = {
		clr_global_flag = dh_waldo_timer
	}

	#They're everywhere!
	option = {
		name = dh_waldo.2.a
		change_religion = waldensian
		add_reform_center = waldensian
		set_global_flag = dh_waldo_timer
		dh_add_neighbor_conv_weakness = yes
	}

}

#Conversion Troubles in [Root.GetName]
province_event = {
	id = dh_waldo.6
	title = dh_waldo.6.t
	desc = dh_waldo.6.desc
	picture = RELIGIOUS_CONVERSION_eventPicture

	trigger = {
		religion = waldensian
		owner = {
			religion = catholic
		}
		has_construction = missionary
		NOT = {
			has_province_modifier = heresy_waldo_6a_modifier
			has_province_modifier = heresy_waldo_6d_modifier
		}
	}

	mean_time_to_happen = {
		months = 240
		modifier = {
			factor = 0.5
			owner = {
				has_estate_privilege = estate_church_enforced_one_faith
			}
		}
		modifier = {
			factor = 0.75
			development = 20
		}
	}

	#And what should we care? They're heretics!
	option = {
		name = dh_waldo.6.a
		ai_chance = {
			factor = 40
			modifier = {
				factor = 40
			}
		}
		spawn_rebels = {
			size = 2
			type = waldensian_rebels
			unrest = 4
		}
		add_province_modifier = {
			name = heresy_waldo_6a_modifier
			duration = 3650
		}
		add_named_unrest = {
			name = heresy_waldo_rebellion_unrest_mod
			value = 8
		}
		owner = {			
			every_neighbor_country = {
				limit = {
					religion = waldensian
				}
				add_opinion = {
					who = ROOT
					modifier = heresy_waldo_6a_opinion_modifier
				}
			}
		}
	}

	#Perhaps we ought to let them be for a while, at least.
	option = {
		name = dh_waldo.6.b
		ai_chance = {
			factor = 60
			modifier = {
				factor = 1.25
				development = 30 
			}
		}
		owner = {
			add_prestige = -5
		}
		add_province_modifier = {
			name = heresy_waldo_6b_modifier
			duration = 3650
		}
		cancel_construction = yes
	}

	#Their heresy is against the laws of this realm. They have no defense.
	option = {
		name = dh_waldo.6.c
		trigger = {
			owner = {
				has_estate_privilege = estate_church_enforced_one_faith
			}
		}
		highlight = yes
		ai_chance = {
			factor = 100
		}
		owner = {
			add_prestige = 15
		}
		spawn_rebels = {
			size = 1
			type = waldensian_rebels
			unrest = 2
		}
		add_province_modifier = {
			name = heresy_waldo_6a_modifier
			duration = 3650
		}
		add_named_unrest = {
			name = heresy_waldo_rebellion_unrest_mod
			value = 8
		}
	}

	#Surely we can come to some sort of agreement over tea! Do... do Waldensians drink tea?
	option = {
		name = dh_waldo.6.d
		trigger = {
			OR = {
				tolerance_to_this = 1
				owner = {
					has_estate_privilege = estate_burghers_enforced_interfaith_dialogue
				}
			}
		}		
		highlight = yes
		ai_chance = {
			factor = 50
			modifier = {
				factor = 1.5
				development = 30
			}
			modifier = {
				factor = 1.5
				province_trade_power = 5
			}
			modifier = {
				factor = 5
				province_with_special_goods_produced_trigger = yes
			}
		}
		add_province_modifier = {
			name = heresy_waldo_6d_modifier
			duration = 3650
		}
		owner = {
			add_prestige = 5
		}
	}

}

#Popularity Among the Poor
country_event = {
	id = dh_waldo.3
	title = dh_waldo.3.t
	desc = dh_waldo.3.desc
	picture = ANGLICAN_SERMON_eventPicture
	goto = dh_waldo_3_target

	trigger = {
		capital_scope = {
			continent = europe
		}
		NOT = {
			religion = waldensian
		}
		any_owned_province = {
			religion = waldensian			
			continent = europe
			NOT = {
				development = 20
				has_province_modifier = dh_waldo_3_a_modifier
				has_province_modifier = dh_waldo_3_b_modifier
			}
		}
	}

	immediate = {
		random_owned_province = {
			limit = {				
				religion = waldensian			
				continent = europe
				NOT = {
					development = 20
					has_province_modifier = dh_waldo_3_a_modifier
					has_province_modifier = dh_waldo_3_b_modifier
				}
			}
			save_event_target_as = dh_waldo_3_target
		}
	}

	#is_triggered_only = yes #on_bi_yearly_pulse_4

	#Changed to on_action
	#Changed back to MTTH

	mean_time_to_happen = {
		months = 180
	}

	#Let them practice in peace.
	option = {
		name = dh_waldo.3.a
		ai_chance = {
			factor = 50
			modifier = {
				factor = 2
				event_target:dh_waldo_3_target = {					
					tolerance_to_this = 2
				}
			}
			modifier = {
				factor = 1.5
				num_of_religion = {
					value = 3
					religion = waldensian
				}
			}
		}
		event_target:dh_waldo_3_target = {			
			add_province_modifier = {
				name = dh_waldo_3_a_modifier
				duration = 3650
			}
		}
	}

	#Crack down on this lay preaching!
	option = {
		name = dh_waldo.3.b
		ai_chance = {
			factor = 50
			modifier = {
				factor = 2
				event_target:dh_waldo_3_target = {		
					NOT = {									
						tolerance_to_this = 2
					}
				}
			}
		}
		event_target:dh_waldo_3_target = {			
			add_province_modifier = {
				name = dh_waldo_3_b_modifier
				duration = 3650
			}
		}
	}

	#Allocate resources to eradicate the people's poverty.
	option = {
		name = dh_waldo.3.c
		ai_chance = {
			factor = 50
		}
		highlight = yes
		trigger = {
			OR = {
				ruler_has_personality = kind_hearted_personality
				ruler_has_personality = tolerant_personality
				has_idea_group = humanist_ideas
			}
		}
		add_years_of_income = -0.25
		add_adm_power = -25
		event_target:dh_waldo_3_target = {			
			add_base_tax = 1
			add_base_production = 1
		}
	}
}

#Prominent Waldensian Preacher

				type = theologian
			}
			employed_advisor = {
				religion = waldensian
				type = inquisitor
			}
		}		
		#Probably don't need this, since the event is now fire by an on_action
		#OR = {
		#	NOT = {
		#		has_country_flag = dh_waldo_4_flag
		#	}
		#	had_country_flag = {
		#		flag = dh_waldo_4_flag
		#		days = 5475
		#	}
		#}
		any_owned_province = {
			development = 10
			religion = waldensian
		}
		OR = {
			religion = catholic
			religion = cathar
			#religion = waldensian
		}
		capital_scope = {
			continent = europe
		}
	}

	immediate = {
		random_owned_province = {
			limit = {
				development = 10
				religion = waldensian
			}
			save_event_target_as = dh_waldo_4_t_target
		}
		set_country_flag = dh_waldo_4_flag
	}

	#is_triggered_only = yes #on_bi_yearly_pulse_4

	#Changed to on_action
	#Changed back to MTTH

	mean_time_to_happen = {
		months = 300
		modifier = {
			factor = 1.5
			employed_advisor = {
				religion = waldensian
				type = theologian
			}
		}
	}

	#We ought to burn him at the stake, not invite him to court!
	option = {
		name = dh_waldo.4.a
		event_target:dh_waldo_4_t_target = {
			add_province_modifier = {
				name = dh_waldo_4_a_modifier
				duration = 3650
			}
		}
		add_country_modifier = {
			name = dh_waldo_4_a_modifier_2
			duration = 3650
		}
	}

	#Take him in. Perhaps his example may inspire some of us to be more pious.
	option = {
		name = dh_waldo.4.b
		add_papal_influence = -5
		event_target:dh_waldo_4_t_target = {
			add_province_modifier = {
				name = dh_waldo_4_b_modifier
				duration = 3650
			}
		}
		define_advisor = {
			type = theologian
			religion = waldensian
			discount = yes
			location = event_target:dh_waldo_4_t_target
			female = no
		}
	}

	#As a holy man, he has no merit. As an inquisitor, though...
	option = {
		name = dh_waldo.4.c
		highlight = yes
		trigger = {
			OR = {
				ruler_has_personality = zealot_personality
				has_idea_group = religious_ideas
				has_idea_group = christian_idea_groups_expanded
				#has_country_flag = dh_waldo_inquisition_declared
			}
		}
		define_advisor = {
			type = inquisitor
			religion = waldensian
			discount = yes
			location = event_target:dh_waldo_4_t_target
			female = no
		}
		event_target:dh_waldo_4_t_target = {
			random = {
				chance = 50
				spawn_rebels = {
					size = 1
					type = waldensian_rebels
				}
			}
		}
	}
}
